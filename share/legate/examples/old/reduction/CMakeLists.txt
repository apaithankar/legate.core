cmake_minimum_required(VERSION 3.22.1 FATAL_ERROR)

project(reduction VERSION 1.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(BUILD_SHARED_LIBS ON)

find_package(legate_core REQUIRED)

legate_add_cpp_subdirectory(src TARGET reduction EXPORT reduction-export)

legate_add_cffi(${CMAKE_CURRENT_SOURCE_DIR}/src/reduction_cffi.h TARGET reduction)
legate_python_library_template(reduction)
legate_default_python_install(reduction EXPORT reduction-export)
