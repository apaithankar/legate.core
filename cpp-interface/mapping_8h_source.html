427 // Work-around for using unique_ptr for PIMPL. unique_ptr requires the type to be defined in
428 // its destructor, as required by delete. This is a problem because the implicitly declared
429 // destructor/move assignment/move constructor calls the unique_ptr destructor, and since we
430 // don't define them, they are implicitly defined inline above.
431 //
432 // One solution then is to manually define these functions out-of-line (can still be done
433 // trivially, i.e. StoreMapping::~StoreMapping() = default), but the whole point of using
434 // unique_ptr is that we *don't* want to write these functions!
435 //
436 // The better solution then is to hide the call to delete behind a custom deleter. Hence
437 // StoreMappingImplDeleter.
438 struct StoreMappingImplDeleter {
439 void operator()(detail::StoreMapping* ptr) const noexcept;
440 };
441
442 std::unique_ptr<detail::StoreMapping, StoreMappingImplDeleter> impl_{};
443};
