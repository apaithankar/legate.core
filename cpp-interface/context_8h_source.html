66class ResourceScope {
67 public:
68 ResourceScope() = default;
69 ResourceScope(int64_t base, int64_t max) : base_(base), max_(max) {}
70
71 public:
72 ResourceScope(const ResourceScope&) = default;
73
74 public:
75 int64_t translate(int64_t local_resource_id) const { return base_ + local_resource_id; }
76 int64_t invert(int64_t resource_id) const
77 {
78 assert(in_scope(resource_id));
79 return resource_id - base_;
80 }
81
82 public:
83 bool valid() const { return base_ != -1; }
84 bool in_scope(int64_t resource_id) const
85 {
86 return base_ <= resource_id && resource_id < base_ + max_;
87 }
88
89 private:
90 int64_t base_{-1};
91 int64_t max_{-1};
92};
