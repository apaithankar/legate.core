62template <typename T, int DIM>
63std::string print_dense_array(const T* base, const Point<DIM>& extents, size_t strides[DIM])
64{
65#ifdef LEGATE_USE_CUDA
66 T* buf = nullptr;
67 if (is_device_only_ptr(base)) {
68 size_t num_elems = 0;
69 for (size_t dim = 0; dim < DIM; ++dim) {
70 num_elems = MAX(num_elems, strides[dim] * extents[dim]);
71 }
72 buf = new T[num_elems];
73 cudaError_t res = cudaMemcpy(buf, base, num_elems * sizeof(T), cudaMemcpyDeviceToHost);
74 assert(res == cudaSuccess);
75 base = buf;
76 }
77#endif // LEGATE_USE_CUDA
78 std::stringstream ss;
79 for (int dim = 0; dim < DIM; ++dim) {
80 if (strides[dim] == 0) continue;
81 ss << "[";
82 }
83 ss << *base;
84 coord_t offset = 0;
85 Point<DIM> point = Point<DIM>::ZEROES();
86 int dim;
87 do {
88 for (dim = DIM - 1; dim >= 0; --dim) {
89 if (strides[dim] == 0) continue;
90 if (point[dim] + 1 < extents[dim]) {
91 ++point[dim];
92 offset += strides[dim];
93 ss << ", ";
94 for (int i = dim + 1; i < DIM; ++i) {
95 if (strides[i] == 0) continue;
96 ss << "[";
97 }
98 ss << base[offset];
99 break;
100 } else {
101 offset -= point[dim] * strides[dim];
102 point[dim] = 0;
103 ss << "]";
104 }
105 }
106 } while (dim >= 0);
107#ifdef LEGATE_USE_CUDA
108 if (buf != nullptr) delete buf;
109#endif // LEGATE_USE_CUDA
110 return ss.str();
111}

122template <int DIM, typename ACC>
123std::string print_dense_array(ACC accessor, const Rect<DIM>& rect)
124{
125 Point<DIM> extents = rect.hi - rect.lo + Point<DIM>::ONES();
126 size_t strides[DIM];
127 const typename ACC::value_type* base = accessor.ptr(rect, strides);
