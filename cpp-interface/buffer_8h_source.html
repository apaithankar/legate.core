73template <typename VAL, int32_t DIM>
74Buffer<VAL, DIM> create_buffer(const Point<DIM>& extents,
75 Memory::Kind kind = Memory::Kind::NO_MEMKIND,
76 size_t alignment = 16)
77{
78 if (Memory::Kind::NO_MEMKIND == kind) kind = find_memory_kind_for_executing_processor(false);
79 auto hi = extents - Point<DIM>::ONES();
80 // We just avoid creating empty buffers, as they cause all sorts of headaches.
81 for (int32_t idx = 0; idx < DIM; ++idx) hi[idx] = std::max<int64_t>(hi[idx], 0);
82 Rect<DIM> bounds(Point<DIM>::ZEROES(), hi);
83 return Buffer<VAL, DIM>(bounds, kind, nullptr, alignment);
84}
