73template <typename VAL, int32_t DIM>
74Buffer<VAL, DIM> create_buffer(const Point<DIM>& extents,
75 Memory::Kind kind = Memory::Kind::NO_MEMKIND,
76 size_t alignment = DEFAULT_ALIGNMENT)
77{
78 static_assert(DIM <= LEGATE_MAX_DIM);
79
80 if (Memory::Kind::NO_MEMKIND == kind) {
81 kind = find_memory_kind_for_executing_processor(false);
82 }
83 auto hi = extents - Point<DIM>::ONES();
84 // We just avoid creating empty buffers, as they cause all sorts of headaches.
85 for (int32_t idx = 0; idx < DIM; ++idx) {
86 hi[idx] = std::max<int64_t>(hi[idx], 0);
87 }
88 return Buffer<VAL, DIM>{Rect<DIM>{Point<DIM>::ZEROES(), std::move(hi)}, kind, nullptr, alignment};
89}
